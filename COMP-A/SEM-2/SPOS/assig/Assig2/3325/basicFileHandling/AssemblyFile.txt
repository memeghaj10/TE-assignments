			START		101
			READ		N
			MOVER		BREG,='1'
			MOVEM		BREG, TERM
AGAIN		MULT		BREG, ='2'
			MOVER		CREG, TERM
			ADD 		CREG, ONE
			MOVEM		CREG, TERM
AGAIN		EQU			200	
			COMP		CREG, N
			ORIGIN		AGAIN + 1
			BC			LE, AGAIN
			MOVEM		BREG, RESULT
			PRINT		RESULT
			STOP
N			DS			1
RESULT		DS			1
ONE 		DC			'1'
TERM		DS			1
			END
